let offlineDB;
const DB_NAME = "quiz_offline_db";
const DB_VERSION = 1;

// Ouvre ou cr√©e la base locale
function openOfflineDB() {
    return new Promise((resolve, reject) => {
        const request = indexedDB.open(DB_NAME, DB_VERSION);

        request.onupgradeneeded = (event) => {
            const db = event.target.result;

            // Table des scores locaux
            if (!db.objectStoreNames.contains("scores")) {
                db.createObjectStore("scores", { keyPath: "id" });
            }

            // File d‚Äôattente pour sync Firestore
            if (!db.objectStoreNames.contains("syncQueue")) {
                db.createObjectStore("syncQueue", { keyPath: "queueId", autoIncrement: true });
            }
        };

        request.onsuccess = (event) => {
            offlineDB = event.target.result;
            resolve();
        };

        request.onerror = () => reject(request.error);
    });
}

// Sauvegarde localement
function saveLocalScore(scoreObj) {
    return new Promise((resolve, reject) => {
        const tx = offlineDB.transaction(["scores"], "readwrite");
        tx.objectStore("scores").put(scoreObj);

        tx.oncomplete = resolve;
        tx.onerror = () => reject(tx.error);
    });
}

// Ajoute un √©l√©ment dans la file d'attente (si offline)
function queueSync(scoreObj) {
    return new Promise((resolve, reject) => {
        const tx = offlineDB.transaction(["syncQueue"], "readwrite");
        tx.objectStore("syncQueue").add(scoreObj);

        tx.oncomplete = resolve;
        tx.onerror = () => reject(tx.error);
    });
}

async function getAllQueuedScores() {
    return new Promise((resolve, reject) => {
        const tx = offlineDB.transaction("syncQueue", "readonly");
        const req = tx.objectStore("syncQueue").getAll();

        req.onsuccess = () => resolve(req.result || []);
        req.onerror = () => reject(req.error);
    });
}

async function clearQueueItem(queueId) {
    return new Promise((resolve, reject) => {
        const tx = offlineDB.transaction("syncQueue", "readwrite");
        tx.objectStore("syncQueue").delete(queueId);

        tx.oncomplete = resolve;
        tx.onerror = () => reject(tx.error);
    });
}

async function getTop10Scores() {
  const tx = offlineDB.transaction("scores", "readonly");
  const req = tx.objectStore("scores").getAll();

  const local = await new Promise((resolve, reject) => {
    req.onsuccess = () => resolve(req.result || []);
    req.onerror = () => reject(req.error);
  });

  let allScores = [...local];

  if (gameState.isAuthReady && Array.isArray(gameState.leaderboard)) {
    const online = gameState.leaderboard.map(x => ({
      userId: x.userId || x.id,
      name: x.name,
      score: x.score,
      timestamp: (x.timestamp?.seconds ? x.timestamp.seconds * 1000 : Date.now())
    }));
    allScores = [...allScores, ...online];
  }

  const bestByUser = new Map();
  for (const s of allScores) {
    if (!s.userId) continue;
    const prev = bestByUser.get(s.userId);
    if (!prev || s.score > prev.score) {
      bestByUser.set(s.userId, s);
    }
  }

  return Array.from(bestByUser.values())
    .sort((a, b) => b.score - a.score || (b.timestamp - a.timestamp))
    .slice(0, 10);
}

        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];

        /**
         * Nettoie une cha√Æne de r√©ponse pour une comparaison de base (minuscule, sans accent ni ponctuation)
         * @param {string} answer - La cha√Æne de r√©ponse.
         * @returns {string} La cha√Æne nettoy√©e.
         */
        function cleanAnswer(answer) {
            // Retire les accents, passe en minuscule, et supprime la plupart des ponctuations
            return answer
                .toLowerCase()
                .normalize('NFD').replace(/[\u0300-\u036f]/g, "")
                .replace(/[.,\/#!$%\^&\*;:{}=\-_`~()]/g, "")
                .trim();
        }

        /**
         * Calcule la distance de Levenshtein (nombre minimal d'√©ditions n√©cessaires) entre deux mots.
         */
        function getEditDistance(a, b) {
            if (a.length === 0) return b.length;
            if (b.length === 0) return a.length;

            const matrix = [];

            for (let i = 0; i <= b.length; i++) {
                matrix[i] = [i];
            }
            for (let j = 0; j <= a.length; j++) {
                matrix[0][j] = j;
            }

            for (let i = 1; i <= b.length; i++) {
                for (let j = 1; j <= a.length; j++) {
                    const cost = a[j - 1] === b[i - 1] ? 0 : 1;
                    matrix[i][j] = Math.min(
                        matrix[i - 1][j] + 1,       // Suppression
                        matrix[i][j - 1] + 1,       // Insertion
                        matrix[i - 1][j - 1] + cost // Substitution
                    );
                }
            }
            return matrix[b.length][a.length];
        }

        function setupLeaderboardListener() {
            if (!gameState.isAuthReady) return;

            const scoresQuery = query(
                getPublicScoresCollectionRef(),
                orderBy('score', 'desc'),
                limit(10) 
            );

            onSnapshot(scoresQuery, (snapshot) => {
                let leaderboard = [];
                snapshot.forEach((doc) => {
                    leaderboard.push(doc.data());
                });

                leaderboard.sort((a, b) => {
                    if (a.score !== b.score) {
                        return b.score - a.score;
                    }
                    const timeA = a.timestamp?.seconds || 0;
                    const timeB = b.timestamp?.seconds || 0;
                    return timeB - timeA;
                });
                
                gameState.leaderboard = leaderboard;
                console.log("Classement mis √† jour.");
                renderApp(); 
            }, (error) => {
                console.error("Erreur d'√©coute du classement:", error);
            });
        }

        function startTimer() {
            // Logique du minuteur dynamique : 10s par 50 caract√®res, minimum 15s
            const currentQ = QUIZ_DATA_FINAL[gameState.currentQIndex];
            const charCount = currentQ.question.length;
            const estimatedLines = Math.ceil(charCount / 50); 
            
            // R√©initialisation de l'initialTimer uniquement si la question change
            if (!gameState.isPaused) {
                gameState.initialTimer = Math.max(15, estimatedLines * 10);
                gameState.timer = gameState.initialTimer;

            gameState.timerInterval = setInterval(() => {
                if (gameState.isPaused) {
                    clearInterval(gameState.timerInterval); // Stop interval si on passe en pause
                    return;
                }

                gameState.timer--;
                const timerDisplay = document.getElementById('timer-display');
                if (timerDisplay) timerDisplay.textContent = gameState.timer;

                const timerRing = document.getElementById('timer-ring');
                if (timerRing) {

        function processAnswer(userAnswer, isTimedOut = false) {
            clearInterval(gameState.timerInterval);
            clearTimeout(quizTimerTimeout); // S'assurer que le d√©lai de 3s est effac√© si l'utilisateur r√©pond vite
            
            const currentQ = QUIZ_DATA_FINAL[gameState.currentQIndex];
            const cleanUserAnswer = cleanAnswer(userAnswer);
            const cleanCorrectAnswer = cleanAnswer(currentQ.answer);

            // Tol√©rance : Max 2 fautes ou 20% de la longueur de la r√©ponse (pour les longues r√©ponses)
            const maxEditDistance = Math.max(1, Math.min(2, Math.ceil(cleanCorrectAnswer.length * 0.2))); 
            const distance = getEditDistance(cleanUserAnswer, cleanCorrectAnswer);

            const isCorrect = distance <= maxEditDistance;
            let pointsAwarded = 0;
            let feedbackTitle = "";
            let feedbackClass = "";
            let feedbackMessage = "";

            // Logique de score sp√©ciale pour "Cascades"
            if (currentQ.section === "Cascades (A-Z)") {
                const requiredLetter = currentQ.requiredLetter.toUpperCase();
                const userFirstLetter = cleanUserAnswer.charAt(0).toUpperCase();
                const startsWithCorrectLetter = userFirstLetter === requiredLetter;

                if (isCorrect && startsWithCorrectLetter) {
                    pointsAwarded = currentQ.points; // +20
                    feedbackTitle = "CASCADE R√âUSSIE !";
                    feedbackClass = "correct-feedback";
                    feedbackMessage = `Excellent ! La r√©ponse <strong>${currentQ.answer}</strong> commence bien par ${requiredLetter}. <br>Vous gagnez ${pointsAwarded} points.`;
                } else {
                    pointsAwarded = -10; // P√©nalit√©
                    feedbackTitle = "P√âNALIT√â (-10 PTS)";
                    feedbackClass = "incorrect-feedback";
                    if (isCorrect && !startsWithCorrectLetter) {
                        feedbackMessage = `La r√©ponse <strong>${currentQ.answer}</strong> est correcte, mais elle ne commence pas par ${requiredLetter}.`;
                    } else if (!isCorrect && startsWithCorrectLetter) {
                        feedbackMessage = `La r√©ponse commence bien par ${requiredLetter}, mais elle est incorrecte. <br>La bonne r√©ponse √©tait : <strong>${currentQ.answer}</strong>.`;
                    } else {
                        feedbackMessage = `R√©ponse incorrecte. La bonne r√©ponse √©tait : <strong>${currentQ.answer}</strong> (devait commencer par ${requiredLetter}).`;
                    }
                }
            } else {
                // Logique normale pour les autres sections
                if (isCorrect) {
                    pointsAwarded = currentQ.points;
                    feedbackTitle = "CORRECT !";
                    feedbackClass = "correct-feedback";
                    feedbackMessage = `${isTimedOut ? "Temps √©coul√©. " : ""} ${isCorrect ? `Tol√©rance accept√©e (Distance: ${distance}). Vous gagnez ${pointsAwarded} points.` : 'Pas de point.'} <br>La bonne r√©ponse √©tait : <strong>${currentQ.answer}</strong>.`;
                } else {
                    pointsAwarded = 0;
                    feedbackTitle = "INCORRECT !";
                    feedbackClass = "incorrect-feedback";
                    feedbackMessage = `${isTimedOut ? "Temps √©coul√©. " : ""} ${isCorrect ? `Tol√©rance accept√©e (Distance: ${distance}). Vous gagnez ${pointsAwarded} points.` : 'Pas de point.'} <br>La bonne r√©ponse √©tait : <strong>${currentQ.answer}</strong>.`;
                }
            }
            
            gameState.currentScore += pointsAwarded;

            // Enregistrer la r√©ponse pour le r√©capitulatif
            gameState.userAnswers.push({
                index: gameState.currentQIndex + 1,
                question: currentQ.question,
                correctAnswer: currentQ.answer,
                userAnswer: userAnswer.trim() || 'N/A (Temps √©coul√©)',
                points: currentQ.points,
                pointsAwarded: pointsAwarded,
                isCorrect: isCorrect,
                section: currentQ.section
            });

            // Si ce n'est pas une cascade, d√©finir les messages de feedback standards (s'ils n'ont pas √©t√© d√©finis par la logique normale)
            if (currentQ.section !== "Cascades (A-Z)") {
                if (!isCorrect && !isTimedOut) {
                    feedbackMessage += `<br><span class="text-sm text-gray-200">Distance d'√©dition: ${distance} (Max tol√©r√©: ${maxEditDistance})</span>`;
                }
            }
            
            // V√©rification de la fin de rubrique avant de passer √† la question suivante
            const checkNextStep = () => {
                gameState.currentQIndex++;

                if (gameState.currentQIndex < QUIZ_DATA_FINAL.length) {
                    
                    const currentSection = QUIZ_DATA_FINAL[gameState.currentQIndex - 1].section;
                    const nextQ = QUIZ_DATA_FINAL[gameState.currentQIndex];

                        showModal(
                            "Rubrique Termin√©e ! üéâ",

        async function getRemoteLeaderboard(max = 10) {
          if (!gameState.isAuthReady) return [];
          const q = query(getPublicScoresCollectionRef(), orderBy('score', 'desc'), limit(max));
          const snap = await getDocs(q);
          const arr = [];
          snap.forEach(docu => {
              const d = docu.data();
              arr.push({
              userId: d.id,
              name: d.name,
              score: d.score,
              timestamp: (d.timestamp?.seconds ? d.timestamp.seconds * 1000 : Date.now())
              });
          });
          return arr;
          }

          async function getCombinedLeaderboard(max = 10) {
          const local = await getLocalLeaderboard(max);
          if (navigator.onLine && gameState.isAuthReady) {
              const remote = await getRemoteLeaderboard(max);
              const map = new Map();
              for (const r of remote) map.set(r.userId, r);
              for (const l of local) {
              const existing = map.get(l.userId);
              if (!existing || l.score > existing.score) {
                  map.set(l.userId, l);
              }
              }
              return Array.from(map.values())
              .sort((a,b) => b.score - a.score || (b.timestamp - a.timestamp))
              .slice(0, max);
          } else {
              return local;
          }
          }
